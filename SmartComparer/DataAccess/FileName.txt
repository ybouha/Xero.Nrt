// accounts.service.ts
import { Injectable } from '@angular/core';
import { HttpClient, HttpErrorResponse } from '@angular/common/http';
import { BehaviorSubject, Observable, throwError } from 'rxjs';
import { catchError, tap } from 'rxjs/operators';

export interface Account {
  id: number;
  name: string;
  amount: number;
  rowVersion?: string;
}

@Injectable({
  providedIn: 'root',
})
export class AccountsService {
  private accountsUrl = '/api/accounts'; // Adjust based on your API endpoint
  private accounts = new BehaviorSubject<Account[]>([]);
  public accounts$ = this.accounts.asObservable();

  constructor(private http: HttpClient) {}

  loadAccounts(): void {
    this.http.get<Account[]>(this.accountsUrl).pipe(
      catchError(this.handleError),
      tap(accounts => this.accounts.next(accounts))
    ).subscribe();
  }

  updateAccount(account: Account): Observable<Account> {
    return this.http.put<Account>(`${this.accountsUrl}/${account.id}`, account).pipe(
      catchError(error => {
        if (error instanceof HttpErrorResponse && error.status === 412) {
          this.loadAccounts(); // Reload the list if a concurrency exception is raised
          return throwError('Concurrency conflict');
        }
        return this.handleError(error);
      })
    );
  }

  private handleError(error: HttpErrorResponse) {
    console.error('An error occurred:', error.error);
    return throwError('An unexpected error occurred.');
  }
}

// accounts.component.ts
import { Component, OnInit, ViewChild } from '@angular/core';
import { AccountsService, Account } from './accounts.service';
import CustomStore from 'devextreme/data/custom_store';
import 'devextreme/data/odata/store';
import { DxDataGridComponent } from 'devextreme-angular';
import { exportDataGrid } from 'devextreme/excel_exporter';
import * as ExcelJS from 'exceljs';
import saveAs from 'file-saver';

@Component({
  selector: 'app-accounts',
  templateUrl: './accounts.component.html',
  styleUrls: ['./accounts.component.css'],
})
export class AccountsComponent implements OnInit {
  @ViewChild(DxDataGridComponent, { static: false }) dataGrid!: DxDataGridComponent;
  dataSource: any;

  constructor(private accountsService: AccountsService) {}

  ngOnInit() {
    this.initializeDataSource();
    this.accountsService.loadAccounts(); // Load accounts initially
  }

  initializeDataSource() {
    this.dataSource = new CustomStore({
      key: 'id',
      load: () => this.accountsService.accounts$.toPromise(),
      insert: (values) => this.accountsService.createAccount(values).toPromise(),
      update: (key, values) => {
        const account = { ...values, id: key };
        return this.accountsService.updateAccount(account).toPromise();
      },
      remove: (key) => this.accountsService.deleteAccount(key).toPromise(),
    });
  }

  onToolbarPreparing(e: any) {
    e.toolbarOptions.items.unshift(
      {
        location: 'after',
        widget: 'dxButton',
        options: {
          icon: 'exportxlsx',
          onClick: () => this.exportGrid(),
        },
      },
      {
        location: 'before',
        widget: 'dxTextBox',
        options: {
          width: 200,
          placeholder: 'Search...',
          onValueChanged: this.applySearch.bind(this),
        },
      }
    );
  }

  applySearch(args: any) {
    this.dataGrid.instance.searchByText(args.value);
  }

  exportGrid() {
    const workbook = new ExcelJS.Workbook();
    const worksheet = workbook.addWorksheet('Accounts');

    exportDataGrid({
      component: this.dataGrid.instance,
      worksheet: worksheet,
      autoFilterEnabled: true,
    }).then(() => {
      workbook.xlsx.writeBuffer().then((buffer) => {
        saveAs(new Blob([buffer], { type: 'application/octet-stream' }), 'Accounts.xlsx');
      });
    });
  }
}
